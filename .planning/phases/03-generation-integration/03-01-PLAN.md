"""Tests for v3 generator module."""
import json
import os
import shutil
from pathlib import Path
from unittest.mock import patch, MagicMock

import pytest

FIXTURES_DIR = Path(__file__).resolve().parent / "fixtures"
GENERATOR_DIR = Path(__file__).resolve().parent.parent.parent / "generator"


@pytest.fixture
def v3_spec():
    with open(FIXTURES_DIR / "synthetic_v3_spec_gen.json") as f:
        return json.load(f)


@pytest.fixture
def output_dir(tmp_path):
    for tmpl in GENERATOR_DIR.glob("*.jinja2"):
        shutil.copy2(tmpl, tmp_path / tmpl.name)
    project_root = GENERATOR_DIR.parent
    shutil.copy2(project_root / "pyproject.toml", tmp_path / "pyproject.toml")
    return tmp_path


def _mock_requests_get(url):
    mock_response = MagicMock()
    mock_response.text = f"# placeholder for {url.split('/')[-1]}\n"
    mock_response.ok = True
    return mock_response


def _run_v3_generation(v3_spec, output_dir):
    import generate_library_v3 as gen_v3
    original_cwd = os.getcwd()
    try:
        os.chdir(output_dir)
        with patch("generate_library_v3.requests.get", side_effect=_mock_requests_get):
            gen_v3.generate_library(
                spec=v3_spec,
                version_number="0.0.0-test",
                is_github_action=False,
            )
    finally:
        os.chdir(original_cwd)


class TestV3GeneratorOutput:
    def test_produces_sync_module(self, v3_spec, output_dir):
        _run_v3_generation(v3_spec, output_dir)
        assert (output_dir / "meraki" / "api" / "networks.py").exists()

    def test_produces_async_module(self, v3_spec, output_dir):
        _run_v3_generation(v3_spec, output_dir)
        assert (output_dir / "meraki" / "aio" / "api" / "networks.py").exists()

    def test_produces_batch_module(self, v3_spec, output_dir):
        _run_v3_generation(v3_spec, output_dir)
        assert (output_dir / "meraki" / "api" / "batch" / "networks.py").exists()

    def test_no_kwargs_update_locals(self, v3_spec, output_dir):
        """GEN-02: No kwargs.update(locals()) in generated code."""
        _run_v3_generation(v3_spec, output_dir)
        for rel_path in [
            "meraki/api/networks.py",
            "meraki/aio/api/networks.py",
            "meraki/api/batch/networks.py",
        ]:
            content = (output_dir / rel_path).read_text()
            assert "kwargs.update(locals())" not in content
            assert "kwargs = locals()" not in content

    def test_batch_action_count(self, v3_spec, output_dir):
        """GEN-04: All batchable actions produce batch methods."""
        _run_v3_generation(v3_spec, output_dir)
        content = (output_dir / "meraki" / "api" / "batch" / "networks.py").read_text()
        expected_count = len(v3_spec["x-batchable-actions"])
        # Count def lines (each batch method = one def)
        method_count = content.count("    def ")
        # Subtract __init__
        method_count -= 1
        assert method_count == expected_count, (
            f"Expected {expected_count} batch methods, got {method_count}"
        )

    def test_explicit_param_construction(self, v3_spec, output_dir):
        """GEN-02: Methods use explicit param list filtering."""
        _run_v3_generation(v3_spec, output_dir)
        content = (output_dir / "meraki" / "api" / "networks.py").read_text()
        # getNetwork has query param 'name', should appear in query_params list
        assert '"name"' in content
        # updateNetwork has body params, should appear in body_params list
        assert "body_params" in content

**generate_library(spec, version_number, is_github_action):**
- Same supported_scopes list as v2 (lines 209-228)
- Call `clear_cache()` at entry (prevents stale cache between runs)
- `batchable_actions = spec["x-batchable-actions"]`
- Set template_dir based on is_github_action
- Create directories: meraki/, meraki/api/, meraki/api/batch/, meraki/aio/, meraki/aio/api/
- Download non-generated files (same logic as v2 lines 258-287), but set `api_version_number = "v3"` for __init__.py
- Call `common.organize_spec(paths, scopes)` (note: this iterates paths dict, which in v3 has path-level params; organize_spec only reads HTTP method keys, which are the nested dicts, so it works)
- Call `generate_modules(spec, batchable_actions, jinja_env, scopes, template_dir)`
- Run ruff format at end

**generate_modules(spec, batchable_actions, jinja_env, scopes, template_dir):**
- Same loop structure as v2 (lines 310-355) but pass `spec` to function generators
- For each scope: open sync/async/batch files, render class templates, call function generators

**generate_standard_and_async_functions(jinja_env, template_dir, section, output, async_output, spec):**
- For each path, methods in section: for each method, endpoint:
  - Get tags, operation (operationId), description (summary)
  - Get path_item from spec["paths"][path] (needed for parse_params_v3)
  - Call `all_params, metadata = parse_params_v3(endpoint, path_item, spec)`
  - Build function definition string from required params + path params (same type-matching logic as v2 lines 377-425, using `return_params(operation, all_params, ["required"])` and `return_params(operation, all_params, ["path"])`)
  - Also catch URL-referenced params not in path_params via regex on path
  - Add pagination params to definition if "perPage" in all_params (same logic as v2 lines 416-422)
  - Add `**kwargs` if `return_params(operation, all_params, ["optional"])` returns anything
  - Build param_descriptions from `return_params(operation, all_params, ["required", "pagination", "optional"])`
  - **CRITICAL GEN-02 FIX:** Set `kwarg_line = ""` (ALWAYS empty string, never kwargs.update(locals()))
  - Build assert_blocks from enum params
  - Get query_params via `return_params(operation, all_params, ["query"])`
  - Get array_params via `return_params(operation, all_params, ["array"])`
  - Get body_params via `return_params(operation, all_params, ["body"])`
  - Get path_params via `return_params(operation, all_params, ["path"])`
  - Ensure all URL-referenced params are in path_params (regex fallback)
  - Determine call_line based on method and params (same logic as v2 parse_get_params/parse_post_and_put_params/parse_delete_params)
  - Render function_template.jinja2 to sync output, async_function_template.jinja2 to async output
  - Pass all template vars including kwarg_line="" (templates handle empty kwarg_line by skipping that block)

**generate_action_batch_functions(jinja_env, template_dir, section, batch_output, batchable_actions, spec):**
- Same structure as v2 (lines 557-710) but:
  - Use parse_params_v3 instead of parse_params
  - Get path_item from spec["paths"][path]
  - Set `kwarg_line = ""` (GEN-02 fix)
  - Match endpoint["description"] OR endpoint["summary"] against batchable_action_summaries (handles v3 inconsistency)
  - Everything else identical: definition building, assert_blocks, template rendering with batch_operation

**Key difference from v2:** The v3 generator never sets kwarg_line to anything other than "". The template's `{% if kwarg_line|length > 0 %}` block will simply never render. The templates already handle explicit param dict construction via `query_params`, `body_params`, `path_params` lists passed as template vars.

**Note on organize_spec compatibility:** The v3 spec paths dict contains path-level "parameters" key alongside HTTP method keys ("get", "put", etc.). organize_spec iterates `for method in methods` which would include "parameters". Must filter: only process keys that are valid HTTP methods. Either:
- Pre-filter paths before passing to organize_spec (strip "parameters" key from each path_item), OR
- Modify the section dict after organize_spec returns

Choose pre-filtering: before calling organize_spec, create a filtered_paths dict that omits the "parameters" key from each path entry. Pass filtered_paths to organize_spec. Keep original spec["paths"] for path_item lookups in generate functions.
